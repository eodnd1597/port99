const word = document.querySelector(".word");
const shadow = e => {
    const { x, y} = e;
    const rect = word.getBoundingClientRect();
    
    const mouseX = (x - rect.left - rect.width/2) / rect.width * 5;
    const mouseY = (y - rect.top - rect.height/2) / rect.height * 5;

    word.style.textShadow =`${mouseX}px ${-mouseY}px 0px rgba(255, 128, 120, 0.8),
                            ${-mouseX}px ${mouseY}px 0px rgba(68, 140, 159, 0.8),
                            ${mouseY}px ${-mouseX}px 0px rgba(36, 158, 29, 0.8),
                            ${-mouseY}px ${mouseX}px 0px rgba(195, 151, 210, 0.8)`;
    console.log(rect)

    // text-shadow: 
    // 10px 10px 0 rgba(255, 128, 120, 0.8),
    // 10px -10px 0 rgba(68, 140, 159, 0.8),
    // -10px 10px 0 rgba(36, 158, 29, 0.8),
    // -10px -10px 0 rgba(195, 151, 210, 0.8);

}
